CREATE TABLE pescadores (
nombre VARCHAR(50) PRIMARY KEY);

CREATE TABLE resultados1 (
    pescador VARCHAR(50) PRIMARY KEY,
    manga INTEGER,
    puntos INTEGER,
    capturas VARCHAR(500),
    FOREIGN KEY (pescador) REFERENCES pescadores(nombre)
);
